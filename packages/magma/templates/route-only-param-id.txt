import { MagmaResponse, type RouteHandler } from '@mahameru/magma';

export const GET: RouteHandler = ({ request, params }) => {
  return MagmaResponse.json({
    success: true,
    method: request.method,
    route: '{{nameKebabCase}}',
    params,
  });
};
